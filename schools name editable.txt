let activeClassId = ""; 

function openCreateMode() {
  // Reset the tracking ID and clear previous form entries
  document.getElementById('edit-class-id').value = "";
  document.getElementById('class-form-title').innerText = "Create New Class";
  document.getElementById('class-form-submit-btn').innerText = "Create Class";
  
  // Clear the actual input text blocks
  document.getElementById('nc-name').value = "";
  document.getElementById('nc-section').value = "";
  document.getElementById('nc-teacher').value = "";
  document.getElementById('student-roster').innerHTML = "";
  
  // Open up the form wrapper visible on screen
  document.getElementById('new-class-form').classList.add('open');
  document.getElementById('roster-section-container').style.display = "block"; 
}

function openEditMode() {
  if (!activeClassId) return;
  
  // Switch titles to reflect an active update session
  document.getElementById('edit-class-id').value = activeClassId;
  document.getElementById('class-form-title').innerText = "Edit Class Details";
  document.getElementById('class-form-submit-btn').innerText = "Save Changes";
  
  // Locate the target class object from your master array
  const currentClass = classesArray.find(c => c.id === activeClassId);
  if (!currentClass) return;
  
  // Populate the text fields with saved system info
  document.getElementById('nc-name').value = currentClass.name || "";
  document.getElementById('nc-section').value = currentClass.section || "";
  document.getElementById('nc-year').value = currentClass.year || "";
  document.getElementById('nc-term').value = currentClass.term || "Third Term";
  document.getElementById('nc-teacher').value = currentClass.teacher || "";
  document.getElementById('nc-school').value = currentClass.school || "";
  
  // Hide the roster adjustments if you only want to focus on editing name/section metadata here
  document.getElementById('roster-section-container').style.display = "none";
  
  // Scroll and open the form panel smoothly
  document.getElementById('new-class-form').classList.add('open');
  document.getElementById('new-class-form').scrollIntoView({ behavior: 'smooth' });
}

function closeClassForm() {
  document.getElementById('new-class-form').classList.remove('open');
  document.getElementById('edit-class-id').value = "";
}

function handleClassFormSubmit() {
  const editId = document.getElementById('edit-class-id').value;
  
  if (editId) {
    // ── UPDATE EXISTNG CLASS METADATA LOGIC ──
    const targetClass = classesArray.find(c => c.id === editId);
    if (targetClass) {
      targetClass.name = document.getElementById('nc-name').value.trim();
      targetClass.section = document.getElementById('nc-section').value.trim();
      targetClass.year = document.getElementById('nc-year').value.trim();
      targetClass.term = document.getElementById('nc-term').value;
      targetClass.teacher = document.getElementById('nc-teacher').value.trim();
      targetClass.school = document.getElementById('nc-school').value.trim();
      
      showNotification("Class updates saved successfully.");
    }
  } else {
    // ── STANDARD NEW CLASS INJECTION LOGIC ──
    // Call your existing build script here...
    showNotification("New class configured successfully.");
  }
  
  closeClassForm();
  renderClassGrid(); // Refresh your dashboard interface cards
}
