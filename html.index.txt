    if (eventTargetBtn) eventTargetBtn.classList.add('active');

  // School Crest Storage Processing Engine via base64 streams
  function handleLogoSelection(event) {
    const file = event.target.files[0];
    if (!file || !file.type.match('image.*')) return;

    const reader = new FileReader();
    reader.onload = function(e) {
      const preview = document.getElementById('logo-preview');
      preview.src = e.target.result;
      preview.style.display = 'block';
      document.getElementById('logo-placeholder').style.display = 'none';
    };
    reader.readAsDataURL(file);
  }

  function saveSchoolProfile() {
    const nameInput = document.getElementById('profile-school-name').value.trim();
    const previewImg = document.getElementById('logo-preview');
    
    if (!nameInput) {
      showNotification("Please state your institution name to verify parameters.");
      return;
    }

    localStorage.setItem('savedSchoolName', nameInput);
    if (previewImg.style.display === 'block' && previewImg.src) {
      localStorage.setItem('savedSchoolLogo', previewImg.src);
    }
    applyProfileToUI(nameInput, localStorage.getItem('savedSchoolLogo'));
    showNotification("School metadata saved successfully.");
  }

  function loadSavedSchoolProfile() {
    const storedName = localStorage.getItem('savedSchoolName');
    const storedLogo = localStorage.getItem('savedSchoolLogo');

    if (storedName) document.getElementById('profile-school-name').value = storedName;
    if (storedLogo) {
      const preview = document.getElementById('logo-preview');
      preview.src = storedLogo;
      preview.style.display = 'block';
      document.getElementById('logo-placeholder').style.display = 'none';
    }
    applyProfileToUI(storedName, storedLogo);
  }

  function applyProfileToUI(name, logoDataUrl) {
    const headerTitle = document.getElementById('global-header-school-name');
    const headerCrest = document.getElementById('global-header-crest');

    if (name) headerTitle.innerText = name;
    if (logoDataUrl) {
      headerCrest.innerHTML = `<img src="${logoDataUrl}" style="width:100%; height:100%; object-fit:cover; border-radius:50%;">`;
    }
  }

  function clearSchoolProfile() {
    if(confirm("Revert profile identity back to global defaults?")) {
      localStorage.removeItem('savedSchoolName');
      localStorage.removeItem('savedSchoolLogo');
      window.location.reload();
    }
  }
