const GRADING = [
  {min:80,grade:'A',letter:'A',remark:'Excellent'},
  {min:70,grade:'B+',letter:'B+',remark:'Very Good'},
  {min:60,grade:'B',letter:'B',remark:'Good'},
  {min:50,grade:'C+',letter:'C+',remark:'Credit'},
  {min:40,grade:'C',letter:'C',remark:'Average'},
  {min:30,grade:'D',letter:'D',remark:'Below Average'},
  {min:20,grade:'E',letter:'E',remark:'Weak'},
  {min:0,grade:'F',letter:'F',remark:'Fail'},
];

function getGrade(score){
  if(score===null||score===undefined||score==='') return {grade:'—',remark:'—'};
  for(const g of GRADING) if(score>=g.min) return g;
  return GRADING[GRADING.length-1];
}
function gradeClass(g){ return g.replace('+','-plus'); }

function loadStudentScores(){
  const cls=document.getElementById('enter-class-select').value;
  if(!cls) return;
  const d=loadData(); const c=d.classes[cls];
  const idx=parseInt(document.getElementById('student-nav').value)||0;
  const stu=c.students[idx];
  const scores=stu.scores||{};

  // Build table rows
  const tbody=document.getElementById('scores-tbody'); tbody.innerHTML='';
  c.subjects.forEach(sub=>{
    const sc=scores[sub]||{sba:'',exam:''};
    const total=(sc.sba!==''&&sc.exam!=='')?Math.round(parseFloat(sc.sba||0)*0.5+parseFloat(sc.exam||0)*0.5):null;
    const g=getGrade(total);
    const tr=document.createElement('tr');
    tr.innerHTML=`
      <td class="subject-name">${sub}</td>
      <td style="text-align:center;"><input type="number" class="score-input" min="0" max="100" value="${sc.sba}" data-sub="${sub}" data-type="sba" oninput="recalcRow(this)"></td>
      <td style="text-align:center;"><input type="number" class="score-input" min="0" max="100" value="${sc.exam}" data-sub="${sub}" data-type="exam" oninput="recalcRow(this)"></td>
      <td class="total-score" id="total-${sub.replace(/\s/g,'_')}">${total!==null?total:'—'}</td>
      <td style="text-align:center;"><div class="grade-badge ${gradeClass(g.grade)}" id="grade-${sub.replace(/\s/g,'_')}">${g.grade}</div></td>
      <td class="remark-text" id="remark-${sub.replace(/\s/g,'_')}">${g.remark}</td>`;
    tbody.appendChild(tr);
  });

  // conduct etc
  document.getElementById('days-present').value=stu.daysPresent||'';
  document.getElementById('days-total').value=stu.daysTotal||'';
  document.getElementById('conduct').value=stu.conduct||'';

  updateGrandTotals();

}

function recalcRow(input){
  const sub=input.dataset.sub;
  const key=sub.replace(/\s/g,'_');
  const row=input.closest('tr');
  const sba=parseFloat(row.querySelector('[data-type="sba"]').value);
  const exam=parseFloat(row.querySelector('[data-type="exam"]').value);
  const total=(!isNaN(sba)&&!isNaN(exam))?Math.round(sba*0.5+exam*0.5):null;
  const g=getGrade(total);
  document.getElementById('total-'+key).textContent=total!==null?total:'—';
  const gb=document.getElementById('grade-'+key);
  gb.textContent=g.grade; gb.className='grade-badge '+gradeClass(g.grade);
  document.getElementById('remark-'+key).textContent=g.remark;
  updateGrandTotals();
}

function updateGrandTotals(){
  const rows=[...document.querySelectorAll('#scores-tbody tr')];
  let sum=0, count=0;
  rows.forEach(tr=>{
    const sba=parseFloat(tr.querySelector('[data-type="sba"]').value);
    const exam=parseFloat(tr.querySelector('[data-type="exam"]').value);
    if(!isNaN(sba)&&!isNaN(exam)){ sum+=Math.round(sba*0.5+exam*0.5); count++; }
  });
  const avg=count?Math.round((sum/count)*10)/10:null;
  document.getElementById('grand-total').textContent=count?sum+'/'+(rows.length*100):'—';
  document.getElementById('grand-avg').textContent=avg!==null?avg.toFixed(1):'—';
  const g=getGrade(avg);
  document.getElementById('overall-grade').textContent=avg!==null?g.grade+' – '+g.remark:'—';
}

}

// ─── RANKING ───
function getStudentScore(stu,subjects){
  if(!stu.scores) return null;
  let sum=0,count=0;
  subjects.forEach(sub=>{
    const sc=stu.scores[sub];
    if(sc&&sc.sba!==''&&sc.exam!==''){
      sum+=Math.round(parseFloat(sc.sba||0)*0.5+parseFloat(sc.exam||0)*0.5);
      count++;
    }
  });
  return count?Math.round((sum/count)*10)/10:null;
}

function getRankedStudents(d,cls){
  const c=d.classes[cls];
  const scored=c.students.map((s,i)=>({...s,_i:i,avg:getStudentScore(s,c.subjects)})).filter(s=>s.avg!==null);
  scored.sort((a,b)=>b.avg-a.avg);
  scored.forEach((s,i)=>{ s._rank=i+1; });
  return scored;
}

function getStudentRank(d,cls,idx){
  const ranked=getRankedStudents(d,cls);
  const s=ranked.find(s=>s._i===idx);
  return s?s._rank:null;
}

function createClass(){
  const name=document.getElementById('nc-name').value.trim();
  if(!name){ toast('Please enter a class name'); return; }
  const subjects=[...document.querySelectorAll('#subjects-checkboxes input:checked')].map(cb=>cb.value);
  if(!subjects.length){ toast('Select at least one subject'); return; }
  const rows=[...document.querySelectorAll('.student-row')];
  const students=rows.map(r=>{ const ins=r.querySelectorAll('input'); return {id:ins[0].value.trim(),name:ins[1].value.trim()}; }).filter(s=>s.name);
  if(!students.length){ toast('Add at least one student'); return; }
  const d=loadData();
  if(d.classes[name]){ toast('A class with this name already exists'); return; }
  d.classes[name]={
    name, subjects, students,
    year:document.getElementById('nc-year').value,
    term:document.getElementById('nc-term').value,
    teacher:document.getElementById('nc-teacher').value,
    school:document.getElementById('nc-school').value,
  };
  saveData(d);
  toggleNewClassForm();
  document.getElementById('nc-name').value='';

  document.getElementById('student-roster').innerHTML=''; _studentCount=0;
  document.getElementById('subjects-checkboxes').innerHTML='';

}

function exportCSV(){
  const cls=document.getElementById('results-class-select').value||document.getElementById('enter-class-select').value;

  const d=loadData(); const c=d.classes[cls];

    const r=ranked.find(s=>s._i===i);
    const subScores=c.subjects.map(sub=>{
      const sc=stu.scores&&stu.scores[sub];
      return (sc&&sc.sba!==''&&sc.exam!=='')?Math.round(parseFloat(sc.sba||0)*0.5+parseFloat(sc.exam||0)*0.5):'';

    const avg=r?r.avg.toFixed(1):'';
    const g=r?getGrade(r.avg).grade:'';
    return [r?r._rank:'',stu.id,stu.name,...subScores,avg,g];
